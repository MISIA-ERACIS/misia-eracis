// MISIA MEMORY ENGINE v2.0 — Google Apps Script
// Bloque 1: Resumen de actuaciones
function generarResumenActuaciones() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const hAct = ss.getSheetByName("ACTUACIONES");
  const hMem = ss.getSheetByName("MEMORIA_DATOS");
  if (!hAct || hAct.getLastRow() < 2) return;
  const data = hAct.getDataRange().getValues().slice(1);
  const totalHoras = data.reduce((s,r) => s + (parseFloat(r[5])||0), 0);
  const finalizadas = data.filter(r => String(r[7]).toUpperCase()==="FINALIZADA").length;
  hMem.appendRow(["TOTAL_ACTUACIONES", data.length, new Date()]);
  hMem.appendRow(["HORAS_TOTALES", totalHoras.toFixed(1), new Date()]);
  hMem.appendRow(["FINALIZADAS", finalizadas, new Date()]);
}

// Bloque 2: Resultados e impacto
function generarResumenResultados() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const hImp = ss.getSheetByName("IMPACTO");
  const hMem = ss.getSheetByName("MEMORIA_DATOS");
  if (!hImp || hImp.getLastRow() < 2) return;
  const data = hImp.getDataRange().getValues().slice(1);
  const rrss = data.filter(r => r[3]).length;
  const prensa = data.filter(r => r[4]).length;
  const impactoVecinal = data.filter(r => r[5]).length;
  hMem.appendRow(["MENCIONES_RRSS", rrss, new Date()]);
  hMem.appendRow(["NOTAS_PRENSA", prensa, new Date()]);
  hMem.appendRow(["IMPACTO_VECINAL", impactoVecinal, new Date()]);
}

// Bloque 3: Inserción sociolaboral
function generarResumenInsercion() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const hIns = ss.getSheetByName("INSERCION");
  const hPart = ss.getSheetByName("PARTICIPANTES");
  const hMem = ss.getSheetByName("MEMORIA_DATOS");
  if (!hIns || !hPart) return;
  const totalPart = hPart.getLastRow() - 1;
  const inserciones = hIns.getLastRow() - 1;
  const pct = totalPart > 0 ? ((inserciones/totalPart)*100).toFixed(1) : "0";
  hMem.appendRow(["TOTAL_INSERCIONES", inserciones, new Date()]);
  hMem.appendRow(["PCT_INSERCION", pct + "%", new Date()]);
}

// Bloque 4: Conclusiones institucionales automáticas
function generarConclusiones() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const hMem = ss.getSheetByName("MEMORIA_DATOS");
  if (!hMem) return;
  const datos = {};
  hMem.getDataRange().getValues().slice(1).forEach(r => { datos[r[0]] = r[1]; });
  const texto = `El presente MIS ha alcanzado un total de ${datos.TOTAL_ACTUACIONES||0} ` +
    `actuaciones con ${datos.HORAS_TOTALES||0} horas de intervención directa. ` +
    `Se ha logrado una tasa de inserción del ${datos.PCT_INSERCION||'0%'}, ` +
    `con un impacto documentado en ${datos.IMPACTO_VECINAL||0} actuaciones ` +
    `de mejora urbana en la zona ERACIS+ de Jaén.`;
  hMem.appendRow(["CONCLUSIONES_AUTO", texto, new Date()]);
}

// Función maestra: genera la memoria completa
function generarMemoriaCompleta() {
  generarResumenActuaciones();
  generarResumenResultados();
  generarResumenInsercion();
  generarConclusiones();
  exportarMemoriaAGoogleDocs();
}

// Exportación a Google Docs + PDF
function exportarMemoriaAGoogleDocs() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const hMem = ss.getSheetByName("MEMORIA_DATOS");
  const datos = {};
  hMem.getDataRange().getValues().slice(1).forEach(r => { datos[r[0]] = r[1]; });
  // Reemplaza {{CAMPO}} en la plantilla de Google Docs
  const templateId = "PEGA_AQUI_ID_DE_TU_PLANTILLA_GOOGLE_DOCS";
  const template = DriveApp.getFileById(templateId);
  const copia = template.makeCopy(`Memoria_MIS_${datos.ID_MIS||'001'}_${new Date().toISOString().slice(0,10)}`);
  const doc = DocumentApp.openById(copia.getId());
  const body = doc.getBody();
  Object.keys(datos).forEach(key => { body.replaceText(`{{${key}}}`, String(datos[key])); });
  doc.saveAndClose();
  // Exportar como PDF
  const pdf = DriveApp.getFileById(copia.getId()).getAs('application/pdf');
  DriveApp.createFile(pdf).setName(`MEMORIA_${datos.ID_MIS||'001'}.pdf`);
}
